A propriedade FileReader.onload contém um manipulador de eventos (event handler) executado quando o evento de carregamento (load)  é ativado, quando o conteúdo lido com  readAsArrayBuffer, readAsBinaryString, readAsDataURL ou readAsText fica disponível.

Exemplo

// Callback de um <input type="file" onchange="onChange(event)">
function onChange(event) {
  var file = event.target.files[0];
  var reader = new FileReader();
  reader.onload = function(event) {
    // O arquivo de texto será impresso aqui
    console.log(event.target.result)
  };

  reader.readAsText(file);
}
